        """
        :return: a new hidden cell of a minesweeper's grid.
                 existence of a bomb, number of bombs in neighborhood
                 have to be stated later.
        :rtype: Cell
        :UC: none
        :Examples:

        >>> cel = Cell()

        >>> cel.is_hypothetic()

        """

        self.__coords = (0, 0)
        self.__bomb = False
        self.__hypo = False
        self.__revealed = False
        self.__nbBombs = 0


    def set_coords(self, x, y) :
        """
        :param x: the abscissa of the Cell
        :type x: int
        :param y: the ordinate of the Cell

        >>> cell.get_coords()
        (10, 3)
        """

        self.__coords = (x, y)

[docs]    def __str__(self):
        """
        :return: a string representation of self state
        :rtype: str
        :UC: none
        :Example:

        >>> cel = Cell()
        >>> str(cel) == ' '
        True
        >>> cel.set_hypothetic()
        >>> str(cel) == '?'
        True
        >>> cel.unset_hypothetic()
        >>> cel.reveal()
        >>> str(cel) == '0'
        True
        >>> cel.incr_number_of_bombs_in_neighborhood()
        >>> str(cel) == '1'
        True
        >>> cel.set_bomb()
        >>> str(cel) == 'B'
        True
        """

        if self.__revealed :
            if self.__bomb :
                return 'B'
            else :
                return str(self.__nbBombs)
        elif self.__hypo :
            return '?'
        else :
            return ' '
